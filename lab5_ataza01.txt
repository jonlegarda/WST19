Lehenik windows.onLoad ekintza erabiliz funtzio bat definitzen dugun, non galderaKop metodoari deitzen diogun.
galderaKop() metodoak laurretik lortu den posta helbidea erabiliz. Gainera setinterval(galderaKop, 20000) zehazten da, 20000 milisegunduro
galderaKOp exekutatu dadin eta beraz, eguneraketak automatikoki egiteko. Dei hau AJAX bidez egiten da. 
 
window.onload = function() {
		galderaKop();
		onlineKopurua(1);
		setInterval(onlineKopurua, 2000);
		setInterval(galderaKop, 20000);
	}
function galderaKop() {
		//alert("sartu da");
		var ePosta = "<?php echo "$_GET[ePosta]";?>";
		xhro3.open("GET", "galderaKopurua.php?ePosta="+ePosta, true);
		xhro3.send();
	}
